51 /// <summary>
52 /// Our extended OVERLAPPED record.
53 /// </summary>
54 /// <remarks>
55 /// The standard OVERLAPPED structure doesn't have any fields for application-specific
56 /// data, so we must extend it.
57 /// </remarks>
58 struct EXTENDED_OVERLAPPED : OVERLAPPED
59 {
60  EXTENDED_OVERLAPPED(LPOVERLAPPED_COMPLETION_ROUTINE func) : data(nullptr), func(func)
61  {
62  memset(this, 0, sizeof(OVERLAPPED));
63  }
64 
65  void *data;
66  LPOVERLAPPED_COMPLETION_ROUTINE func;
67  io_scheduler *m_scheduler;
68 };

196  pExtOverlapped->func(result, (DWORD)numberOfBytesTransferred, (LPOVERLAPPED)pOverlapped);
