130  static void* thread_start(void *arg)
131  {
132 #if (defined(ANDROID) || defined(__ANDROID__))
133  // Spinlock on the JVM calling JNI_OnLoad()
134  while (JVM == nullptr)
135  {
136  pplx::details::platform::YieldExecution();
137  }
138  // Calling get_jvm_env() here forces the thread to be attached.
139  get_jvm_env();
140  pthread_cleanup_push(detach_from_java, nullptr);
141 #endif
142  threadpool* _this = reinterpret_cast<threadpool*>(arg);
143  try
144  {
145  _this->m_service.run();
146  }
147  catch (const _cancel_thread&)
148  {
149  // thread was cancelled
150  }
151  catch (...)
152  {
153  // Something bad happened
154 #if (defined(ANDROID) || defined(__ANDROID__))
155  // Reach into the depths of the 'droid!

160 #endif
161  }

163  pthread_cleanup_pop(true);
164 #endif

166  }
